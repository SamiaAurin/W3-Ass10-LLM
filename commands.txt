************** Create The App if you do not want to clone it *******************

pip install django psycopg2
django-admin startproject ollama_project 
python manage.py startapp property_info

*************************************************************************************************************

******************* Create a network to share the travelscraper-db-1 and ollamadb-1 *************************

docker network create shared-network

*********************************************************************
******************* See the tables of ollama_data_new **************

docker exec -it ollama-db-new psql -U username -d ollama_data_new

SELECT * FROM property_summary LIMIT 2;
SELECT * FROM property_rating_review LIMIT 2;

**************************************
 /////////// Super User Creation /////////////
docker exec -it django-new python manage.py createsuperuser

/////////// To Migrate /////////////
docker exec -it django-new python manage.py makemigrations
docker exec -it django-new python manage.py migrate

///////////  CLI FILES /////////////
docker exec -it django-new python manage.py rewrite_property_titles
docker exec -it django-new python manage.py rewrite_property_summary
docker exec -it django-new python manage.py rewrite_property_rating_review


************************ If you want to restart only the django container ********************************
docker-compose restart django-new

************************ To pull any model and see the list *****************************
docker-compose exec ollama /bin/bash
ollama pull tinyllama  or ollama pull phi
- To check the list of models run:
ollama list


************************ For Testing **************************
docker exec -it django-new bash
pip install coverage
coverage run --source='.' manage.py test
coverage report
